CREATE TABLE  "CUSTOMER" 
   (	"ID_CUSTOMER" VARCHAR2(5) NOT NULL ENABLE, 
	"NOMOR_KTP" NUMBER(15,0), 
	"NAMA_CUSTOMER" VARCHAR2(55), 
	"NOMOR_TELEPON" VARCHAR2(13), 
	"PEKERJAAN" VARCHAR2(55), 
	"ALAMAT" VARCHAR2(155), 
	"JENIS_KELAMIN" VARCHAR2(12), 
	"PASSWORD" VARCHAR2(256) NOT NULL ENABLE, 
	"USERNAME" VARCHAR2(56) NOT NULL ENABLE, 
	 CONSTRAINT "CUSTOMER_PK" PRIMARY KEY ("ID_CUSTOMER") ENABLE
   ) ;

CREATE OR REPLACE TRIGGER  "BI_CUSTOMER" 
  before insert on "CUSTOMER"               
  for each row  
begin   
  if :NEW."ID_CUSTOMER" is null then 
    select "CUSTOMER_SEQ".nextval into :NEW."ID_CUSTOMER" from dual; 
  end if; 
end; 

/
ALTER TRIGGER  "BI_CUSTOMER" ENABLE;
