CREATE TABLE  "BARANG" 
   (	"ID_BARANG" VARCHAR2(5) NOT NULL ENABLE, 
	"ID_JENIS_BARANG" VARCHAR2(5) NOT NULL ENABLE, 
	"NAMA_BARANG" VARCHAR2(55), 
	"KUALITAS" VARCHAR2(1), 
	 CONSTRAINT "BARANG_PK" PRIMARY KEY ("ID_BARANG") ENABLE
   ) ;ALTER TABLE  "BARANG" ADD CONSTRAINT "BARANG_FK" FOREIGN KEY ("ID_JENIS_BARANG")
	  REFERENCES  "JENIS_BARANG" ("ID_JENIS_BARANG") ENABLE;

CREATE OR REPLACE TRIGGER  "BI_BARANG" 
  before insert on "BARANG"               
  for each row  
begin   
  if :NEW."ID_BARANG" is null then 
    select "BARANG_SEQ".nextval into :NEW."ID_BARANG" from dual; 
  end if; 
end; 

/
ALTER TRIGGER  "BI_BARANG" ENABLE;
